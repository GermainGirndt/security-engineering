rm:
a) löscht Dateien, aber standartmäßig keine Directories
b) /bin/rm
c) rm text.doc
d) unlink()

mv:
a) verschiebt Dateien oder benennt sie um
b) /bin/mv
c) mv text.doc Documents
d) rename()

chmod:
a) ändert die Berechtigungen einer Datei oder eines Verzeichnisses
b) /bin/chmod
c) chmod 700 Documents
d) fchmod()

chown:
a) ändert den Besitzer und Gruppe einer Datei
b) /bin/chown
c) chown ametz text.doc
d) fchown()

mkdir:
a) erstellt eine leeres Verzeichnis
b) /bin/mkdir
c) mkdir NewDirectory
d) mkdir()

rmdir:
a) löscht ein leeres Verzeichnis
b) /bin/rmdir
c) rmdir NewDirectory
d) rmdir()

kill:
a) sendet ein Signal an einen Prozess
b) /bin/kill
c) kill -15 909808
d) kill()
